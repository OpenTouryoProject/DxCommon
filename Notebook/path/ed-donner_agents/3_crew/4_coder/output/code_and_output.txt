# 1 - 1/3 + 1/5 - 1/7 + ... の最初の 10,000 項を計算し、合計を 4 倍する

def calculate_series(n_terms):
    total = 0.0
    for n in range(n_terms):
        term = (-1) ** n / (2 * n + 1)  # 交互級数の計算
        total += term
    return total * 4  # 合計を4倍する

# 最初の 10,000 項を計算
result = calculate_series(10000)
print(result)

# 出力
# 3.1414926535900345
